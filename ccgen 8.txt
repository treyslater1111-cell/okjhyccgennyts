    private fun startQuantumFeed() {
        GlobalScope.launch(Dispatchers.IO) {
            while (quantumEntropyAvailable) {
                val entropy = quantumGenerator.getQuantumEntropy()
                val randomCard = generateCardWithQuantumEntropy(entropy)
                
                withContext(Dispatchers.Main) {
                    quantumDisplay.text = "Quantum Bits: ${entropy.take(64)}..."
                    
                    // Auto-add quantum-generated cards
                    if (quantumToggle.isChecked) {
                        addQuantumCard(randomCard)
                    }
                }
                
                delay(1000)
            }
        }
    }
    
    private fun generateCardWithQuantumEntropy(entropy: String): CardData {
        // Use quantum entropy for truly random generation
        val hash = MessageDigest.getInstance("SHA-256").digest(entropy.toByteArray())
        val hexHash = hash.joinToString("") { "%02x".format(it) }
        
        // Generate card from quantum hash
        val bin = hexHash.substring(0, 6).filter { it.isDigit() }.take(6)
        val number = generateLuhnFromQuantum(bin, hexHash)
        
        return CardData(
            number = number,
            expiry = generateQuantumExpiry(hexHash),
            cvv = hexHash.substring(20, 23),
            name = "QUANTUM_${hexHash.substring(30, NA)}",
            bin = bin,
            track1 = generateQuantumTrack1(number, hexHash),
            track2 = generateQuantumTrack2(number, hexHash),
            bank = "Quantum Bank",
            type = "QUANTUM",
            country = "XX",
            valid = true,
            balance = hexHash.substring(40, 45).toDoubleOrNull() ?: 1000.0,
            quantumEntropy = entropy,
            quantumGenerated = true
        )
    }

    private fun generateEthereumWallet() {
        GlobalScope.launch(Dispatchers.IO) {
            try {
                // Generate wallet
                val walletFile = WalletUtils.generateNewWalletFile(
                    blockchainEngine.walletPassword,
                    blockchainEngine.walletDirectory,
                    true
                )
                
                val credentials = WalletUtils.loadCredentials(
                    blockchainEngine.walletPassword,
                    File(blockchainEngine.walletDirectory, walletFile)
                )
                
                val address = credentials.address
                val privateKey = credentials.ecKeyPair.privateKey.toString(16)
                
                // Fund with testnet ETH
                val web3 = Web3j.build(HttpService("https://sepolia.infura.io/v3/YOUR-API-KEY"))
                val transfer = Transfer.sendFunds(web3, credentials, address, BigDecimal.ONE, Convert.Unit.ETHER)
                
                val receipt = transfer.send()
                
                withContext(Dispatchers.Main) {
                    blockchainLog.text = """
                        === ETHEREUM WALLET GENERATED ===
                        Address: ${address}
                        Private Key: ${privateKey.take(16)}...
                        Transaction: ${receipt.transactionHash}
                        Balance: 1.0 ETH (Testnet)
                        
                        Store private key securely!
                    """.trimIndent()
                    
                    logToSystem("ETH Wallet: $address")
                    
                    // Save to encrypted storage
                    saveWalletToVault(address, privateKey, "ETHEREUM")
                }
                
            } catch (e: Exception) {
                logToSystem("Wallet error: ${e.message}")
            }
        }
    }
