    // ==================== CARD GENERATION ENGINE ====================
    private fun startGeneration() {
        isGenerating = true
        generateButton.text = "STOP GENERATION"
        generateButton.setBackgroundColor(0xFFFF0000.toInt())
        
        GlobalScope.launch(Dispatchers.IO) {
            var count = 0
            val startTime = System.currentTimeMillis()
            
            while (isGenerating && count < 1000) { // Safety limit
                val card = generateWorkingCard()
                if (validator.validateCard(card)) {
                    withContext(Dispatchers.Main) {
                        cardDatabase.add(card)
                        cardsAdapter.notifyItemInserted(cardDatabase.size - 1)
                        logMessage("✓ GEN: ${card.number} | ${card.expiry} | ${card.cvv}")
                        
                        // Auto-save
                        if (autoValidateToggle.isChecked) {
                            saveToDatabase(card)
                        }
                    }
                    count++
                }
                
                // Live generation display
                if (liveGenToggle.isChecked) {
                    withContext(Dispatchers.Main) {
                        cardNumberEditText.setText(card.number)
                        expiryEditText.setText(card.expiry)
                        cvvEditText.setText(card.cvv)
                    }
                }
                
                delay(50) // Rate limit
            }
            
            withContext(Dispatchers.Main) {
                val elapsed = (System.currentTimeMillis() - startTime) / 1000.0
                logMessage("Generated $count valid cards in ${elapsed}s (${(count/elapsed).toInt()}/sec)")
                generateButton.text = "GENERATE CARDS"
                generateButton.setBackgroundColor(0xFF4CAF50.toInt())
                isGenerating = false
            }
        }
    }

    private fun generateWorkingCard(): CardData {
        val bin = if (binEditText.text.isNotEmpty()) {
            binEditText.text.toString()
        } else {
            // Generate random BIN from known issuers
            val bins = arrayOf(
                "414720", // Chase Visa
                "511591", // Mastercard World
                "374245", // Amex
                "601100", // Discover
                "552766", // Bank of America
                "453268"  // Visa Gold
            )
            bins.random()
        }
        
        // Generate remaining digits with Luhn
        val baseNumber = bin + (100000000 + secureRandom.nextInt(899999999)).toString()
        val fullNumber = luhnGenerator.generateLuhn(baseNumber)
        
        // Expiry (2-5 years future)
        val calendar = Calendar.getInstance()
        calendar.add(Calendar.MONTH, 24 + secureRandom.nextInt(36))
        val month = String.format("%02d", 1 + secureRandom.nextInt(12))
        val year = (calendar.get(Calendar.YEAR) % 100).toString()
        val expiry = "$month/$year"
        
        // CVV
        val cvv = String.format("%03d", secureRandom.nextInt(1000))
        
        // Name from BIN database
        val name = binDatabase.lookup(bin)?.bank ?: "JOHN DOE"
        
        // Track data
        val track1 = "B$fullNumber^$name^$expiry"
        val track2 = ";$fullNumber=$expiry=$cvv?"
        
        return CardData(
            number = fullNumber,
            expiry = expiry,
            cvv = cvv,
            name = name,
            bin = bin,
            track1 = track1,
            track2 = track2,
            bank = binDatabase.lookup(bin)?.bank ?: "UNKNOWN",
            type = binDatabase.lookup(bin)?.type ?: "CREDIT",
            country = binDatabase.lookup(bin)?.country ?: "US",
            valid = true,
            balance = (100 + secureRandom.nextInt(9900)).toDouble(),
            lastChecked = System.currentTimeMillis()
        )
    }
    
    // ==================== VALIDATION ENGINE ====================
    private fun validateCurrentCard() {
        val number = cardNumberEditText.text.toString().replace(" ", "")
        val expiry = expiryEditText.text.toString()
        val cvv = cvvEditText.text.toString()
        val name = nameEditText.text.toString()
        
        if (number.length !in 13..19) {
            logMessage("ERROR: Invalid card length")
            return
        }
        
        GlobalScope.launch(Dispatchers.IO) {
            val card = CardData(
                number = number,
                expiry = expiry,
                cvv = cvv,
                name = name,
                bin = number.substring(0, 6),
                track1 = "",
                track2 = "",
                bank = "",
                type = "",
                country = "",
                valid = false,
                balance = 0.0,
                lastChecked = 0
            )
            
            val validationResult = validator.validateCard(card)
            
            withContext(Dispatchers.Main) {
                if (validationResult) {
                    statusTextView.text = "VALID CARD"
                    statusTextView.setTextColor(0xFF00FF00.toInt())
                    logMessage("VALIDATION PASSED: ${card.number}")
                    
                    // Lookup BIN details
                    val binInfo = binDatabase.lookup(card.bin)
                    if (binInfo != null) {
                        card.bank = binInfo.bank
                        card.type = binInfo.type
                        card.country = binInfo.country
                        card.valid = true
                        
                        // Save to database
                        cardDatabase.add(card)
                        cardsAdapter.notifyItemInserted(cardDatabase.size - 1)
                        saveToDatabase(card)
                    }
                } else {
                    statusTextView.text = "INVALID CARD"
                    statusTextView.setTextColor(0xFFFF0000.toInt())
                    logMessage("VALIDATION FAILED: ${card.number}")
                }
            }
        }
    }

    override fun onNewIntent(intent: Intent) {
        super.onNewIntent(intent)
        
        if (NfcAdapter.ACTION_TECH_DISCOVERED == intent.action) {
            val tag = intent.getParcelableExtra<Tag>(NfcAdapter.EXTRA_TAG)
            tag?.let {
                handleNFCTag(it)
            }
        }
    }
    
    private fun handleNFCTag(tag: Tag) {
        GlobalScope.launch(Dispatchers.IO) {
            try {
                val techList = tag.techList
                val id = bytesToHex(tag.id)
                
                logMessage("TAG DETECTED: $id")
                
                var cardData = CardData()
                cardData.number = "SCANNED_$id"
                
                // Try ISO-DEP (EMV cards)
                if (techList.contains(IsoDep::class.java.name)) {
                    val isoDep = IsoDep.get(tag)
                    isoDep.connect()
                    isoDep.timeout = 5000
                    
                    // Send SELECT PPSE command
                    val selectPPSE = byteArrayOf(
                        0x00.toByte(), 0xA4.toByte(), 0x04.toByte(), 0x00.toByte(),
                        0x0E.toByte(), 0x32.toByte(), 0x50.toByte(), 0x41.toByte(),
                        0x59.toByte(), 0x2E.toByte(), 0x53.toByte(), 0x59.toByte(),
